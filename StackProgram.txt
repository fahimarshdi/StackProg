#include <iostream>
using namespace std;
const int MAX_ITEMS = 5;
struct stack
{
int items [MAX_ITEMS];
int top;
};
void makeEmpty(stack &s)
{
s.top = -1; //(no items in the stack)
}
bool isfull (stack s)
{
if (s.top==MAX_ITEMS-1)
return true;
else
return false;
}
bool isempty (stack s)
{
return(s.top==MAX_ITEMS-1);
}
void push (stack &s, int item)
{
if (isfull(s))
cout<<"OVERFLOW\n";
s.top++;
s.items[s.top]=item;
}
int pop (stack &s)
{
if (isempty(s))
{
cout<<"stack is EMPTY\n";
return 0;
}
else
s.top--;
return s.items[s.top+1];
}
